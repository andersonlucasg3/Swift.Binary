object:

object {
    1 - type | 
    1 - array (false) | 
    4 - key name length | 
    n - key content | 
    4 - number of keys | 
    n - keys and values
} 

object array {
    1 - type | 
    1 - array (true) | 
    4 - key name length | 
    n - key content | 
    4 - number of objects | 
    [object...]
}

primitive types:

primitive {
    1 - type | 
    1 - array (false) | 
    4 - key name length | 
    n - key content | 
    n - value
}

primitive array {
    1 - type | 
    1 - array (true) | 
    4 - key name length | 
    n - key content | 
    4 - number of values | 
    [value]
}
